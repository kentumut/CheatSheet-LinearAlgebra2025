\subsection{Computing the three fundamental subspaces}
\tsThe{4.31 (Basis of \(C(A)\): Pivots columns of $\operatorname{RREF}$)}
{\(R\) is RREF of \(A\), then all columns at pivots of \(R\) form a basis of \(C(A)\): \(\dim(C(A)) = \text{rank}(A) = r\)}
\newline
\tsThe{4.32 (Basis of \(R(A)\): Nonzero rows of $\operatorname{RREF}(A)$)}
{Nonzero rows of $\operatorname{RREF}(A)$ form a basis of
    $R(A)$, so, $\dim(R(A)) = r$.
}
\newline
\tsThe{4.33 (Row rank equals columns rank)}{\(\text{rank}(A) = \text{rank}(A^\top)\)}
\newline
\tsCor{4.34 (Rank is at most min of the matrix dimensions)}{\(A\) is a \(m \times n\) matrix with rank \(r\) \(\Rightarrow\) \(r \leq \min(n,m)\).}
\newline
\tsLem{4.35 (Nullspace isomorphism)}{$R = \operatorname{RREF}(A)$, then \(T : N(R) \to \tsS{R}^{n-r}\) is an isomorphism between \(N(R)\) and \(\tsS{R}^{n-r}\) \(\Rightarrow\) \(\dim(N(R)) = n - r\).}
\newline
\tsThe{4.36 (Basis of \(N(A)\): Non-pivot columns of $\operatorname{RREF}(A)$)}
{If $\operatorname{rank}(A)=r$, then \(\dim(N(A))=n-r.\)
\newline
For finding a basis of N(A):
First put A into RREF form R.
Identify the pivot columns and the non-pivot columns. Write the system Ax = 0.
Write the pivot variables in terms of the non-pivot (free) variables.
Then, for each non-pivot column, set the corresponding free variable to 1
and all other free variables to 0, and compute the resulting vector.
These vectors form a basis of N(A).
}
